# Tracker — TOML Spec Export Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add `GET /api/experiments/:id/spec.toml` that returns a complete, self-contained TOML document for an experiment — resolving metric/population/owner slugs into full objects — plus a "Copy TOML" button in the sidebar.

**Architecture:** One new V function `build_experiment_toml(exp, metrics, populations, owners)` builds the TOML string from resolved catalog objects. The route handler fetches all four tables and calls it. The frontend fetches and copies to clipboard. Export-only (no import).

**Tech Stack:** V / veb / SQLite (backend), Alpine.js + Clipboard API (frontend), no new deps.

## Global Constraints

- All backend code goes in `apps/tracker/main.v` (single-file convention).
- Routes: `@['/path'; method]` attribute + `pub fn (app &App) handler(mut ctx Context, ...) veb.Result`.
- TOML is built as a plain string — V has no stdlib TOML encoder; hand-build it.
- Escape double-quotes in TOML strings: replace `"` with `\"`.
- Kill stale server before testing: `pkill -f apps/tracker && pkill -f "v run ." ; sleep 1 && v run . &`
- Frontend static files (`index.html`, `app.css`) update on browser refresh. `app.js` changes require V recompile.

- [ ] **Step 2: Add `toml_escape` helper and `build_experiment_toml` function**

Add these two functions immediately before `get_current_time()`:

```v
fn toml_escape(s string) string {
	return s.replace('\\', '\\\\').replace('"', '\\"').replace('\n', '\\n')
}

fn build_experiment_toml(exp Experiment, metrics []Metric, populations []Population, owners []Owner) string {
	mut t := '# Experiment Spec — generated by Experiment Tracker\n\n'

	// ── [experiment] ──────────────────────────────────────────────
	t += '[experiment]\n'
	t += 'id                 = ${exp.id}\n'
	t += 'name               = "${toml_escape(exp.name)}"\n'
	t += 'description        = "${toml_escape(exp.description)}"\n'
	t += 'hypothesis         = "${toml_escape(exp.hypothesis)}"\n'
	t += 'test_type          = "${toml_escape(exp.test_type)}"\n'
	t += 'status             = "${toml_escape(exp.status)}"\n'
	t += 'product_area       = "${toml_escape(exp.product_area)}"\n'
	t += 'created_at         = "${toml_escape(exp.created_at)}"\n'
	t += 'started_at         = "${toml_escape(exp.started_at)}"\n'
	t += 'completed_at       = "${toml_escape(exp.completed_at)}"\n'
	if exp.outcome != '' {
		t += 'outcome            = "${toml_escape(exp.outcome)}"\n'
	}
	if exp.notes != '' {
		t += 'notes              = "${toml_escape(exp.notes)}"\n'
	}
	t += '\n'

	// ── [owner] ───────────────────────────────────────────────────
	mut owner_found := false
	for o in owners {
		if o.name == exp.owner {
			t += '[owner]\n'
			t += 'name  = "${toml_escape(o.name)}"\n'
			t += 'email = "${toml_escape(o.email)}"\n'
			t += 'team  = "${toml_escape(o.team)}"\n'
			t += '\n'
			owner_found = true
			break
		}
	}
	if !owner_found && exp.owner != '' {
		t += '[owner]\n'
		t += 'name = "${toml_escape(exp.owner)}"\n'
		t += '\n'
	}

	// ── [design] ──────────────────────────────────────────────────
	t += '[design]\n'
	t += 'statistical_test    = "${toml_escape(exp.statistical_test)}"\n'
	t += 'power               = ${exp.power}\n'
	t += 'significance_level  = ${exp.significance_level}\n'
	t += 'mde                 = "${toml_escape(exp.mde)}"\n'
	t += 'duration_estimate   = "${toml_escape(exp.duration_estimate)}"\n'
	t += 'sample_size_per_group = ${exp.sample_size_per_group}\n'
	t += '\n'

	// ── [primary_metric] ──────────────────────────────────────────
	if exp.primary_metric_slug != '' {
		t += '[primary_metric]\n'
		t += 'slug = "${toml_escape(exp.primary_metric_slug)}"\n'
		for m in metrics {
			if m.slug == exp.primary_metric_slug {
				t += 'name         = "${toml_escape(m.name)}"\n'
				t += 'metric_type  = "${toml_escape(m.metric_type)}"\n'
				if m.baseline_value != 0 {
					t += 'baseline     = ${m.baseline_value}\n'
				}
				break
			}
		}
		t += '\n'
	}

	// ── [[variants]] ──────────────────────────────────────────────
	variants_raw := exp.variants
	if variants_raw != '' && variants_raw != '[]' {
		t += '# Variants (raw JSON — parse as needed)\n'
		t += 'variants_json = "${toml_escape(variants_raw)}"\n'
		t += '\n'
	}

	// ── [[populations]] ───────────────────────────────────────────
	pop_slugs_raw := exp.population_slugs
	if pop_slugs_raw != '' && pop_slugs_raw != '[]' {
		pop_slugs := json.decode([]string, pop_slugs_raw) or { []string{} }
		for slug in pop_slugs {
			t += '[[populations]]\n'
			t += 'slug = "${toml_escape(slug)}"\n'
			for p in populations {
				if p.slug == slug {
					t += 'name = "${toml_escape(p.name)}"\n'
					if p.population_count > 0 {
						t += 'count = ${p.population_count}\n'
					}
					break
				}
			}
			t += '\n'
		}
	}

	// ── [[guardrail_metrics]] ─────────────────────────────────────
	guard_raw := exp.guardrail_slugs
	if guard_raw != '' && guard_raw != '[]' {
		guard_slugs := json.decode([]string, guard_raw) or { []string{} }
		for slug in guard_slugs {
			t += '[[guardrail_metrics]]\n'
			t += 'slug = "${toml_escape(slug)}"\n'
			for m in metrics {
				if m.slug == slug {
					t += 'name        = "${toml_escape(m.name)}"\n'
					t += 'metric_type = "${toml_escape(m.metric_type)}"\n'
					break
				}
			}
			t += '\n'
		}
	}

	// ── [[learning_metrics]] ──────────────────────────────────────
	learn_raw := exp.learning_metric_slugs
	if learn_raw != '' && learn_raw != '[]' {
		learn_slugs := json.decode([]string, learn_raw) or { []string{} }
		for slug in learn_slugs {
			t += '[[learning_metrics]]\n'
			t += 'slug = "${toml_escape(slug)}"\n'
			for m in metrics {
				if m.slug == slug {
					t += 'name = "${toml_escape(m.name)}"\n'
					break
				}
			}
			t += '\n'
		}
	}

	// ── [instrumentation] ─────────────────────────────────────────
	if exp.flag_key != '' || exp.event_lineage != '' {
		t += '[instrumentation]\n'
		if exp.flag_key != '' {
			t += 'flag_key      = "${toml_escape(exp.flag_key)}"\n'
		}
		if exp.event_lineage != '' {
			t += 'event_lineage = "${toml_escape(exp.event_lineage)}"\n'
		}
		t += '\n'
	}

	return t
}
```

- [ ] **Step 3: Add the route handler**

Add immediately after `build_experiment_toml` (before `get_current_time`):

```v
@['/api/experiments/:id/spec.toml'; get]
pub fn (app &App) experiment_spec_toml(mut ctx Context, id int) veb.Result {
	exps := sql app.db {
		select from Experiment where id == id
	} or {
		return ctx.server_error('db error')
	}
	if exps.len == 0 {
		return ctx.not_found()
	}
	metrics := sql app.db {
		select from Metric
	} or { []Metric{} }
	populations := sql app.db {
		select from Population
	} or { []Population{} }
	owners := sql app.db {
		select from Owner
	} or { []Owner{} }
	toml_str := build_experiment_toml(exps[0], metrics, populations, owners)
	ctx.res.set_header(.content_type, 'text/plain; charset=utf-8')
	return ctx.text(toml_str)
}
```

- [ ] **Step 1: Add `copyToml` and `downloadToml` to the Alpine store**

In `Alpine.store('app', {...})` in `app.js`, add after `deleteExp`:

```js
async copyToml(exp) {
    if (!exp?.id) return;
    try {
        const text = await fetch(`/api/experiments/${exp.id}/spec.toml`).then(r => r.text());
        await navigator.clipboard.writeText(text);
        this.toast('TOML copied to clipboard');
    } catch (e) {
        this.toast('Copy failed: ' + e.message, 'error');
    }
},

async downloadToml(exp) {
    if (!exp?.id) return;
    try {
        const text = await fetch(`/api/experiments/${exp.id}/spec.toml`).then(r => r.text());
        const slug = (exp.name || 'experiment').toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-+|-+$/g, '');
        const a = document.createElement('a');
        a.href = 'data:text/plain;charset=utf-8,' + encodeURIComponent(text);
        a.download = `${slug}.toml`;
        a.click();
    } catch (e) {
        this.toast('Download failed: ' + e.message, 'error');
    }
},
```
